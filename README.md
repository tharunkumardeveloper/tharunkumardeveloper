Create this file: **`.github/workflows/chess.yml`**

```yaml
name: Chess Game
on:
  issues:
    types: [opened]
  workflow_dispatch:

jobs:
  play_chess:
    if: startsWith(github.event.issue.title, 'chess:')
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install chess library
        run: pip install chess

      - name: Process chess move
        run: |
          python .github/scripts/chess_move.py "${{ github.event.issue.title }}"

      - name: Commit updated board
        run: |
          git config --global user.name "Chess Bot 🤖"
          git config --global user.email "chess-bot@tharun.dev"
          git add README.md
          git diff --staged --quiet || git commit -m "♟️ Chess move: ${{ github.event.issue.title }}"
          git push

      - name: Close issue
        uses: peter-evans/close-issue@v3
        with:
          comment: "♟️ Move recorded! The board has been updated. Check the README to see the new position. Good game!"
```

Then create **`.github/scripts/chess_move.py`**:

```python
import chess
import re
import sys

PIECES = {
    'R': '♖', 'N': '♘', 'B': '♗', 'Q': '♕', 'K': '♔', 'P': '♙',
    'r': '♜', 'n': '♞', 'b': '♝', 'q': '♛', 'k': '♚', 'p': '♟',
    '.': '·'
}

def board_to_markdown(board):
    lines = ['| | a | b | c | d | e | f | g | h |']
    lines.append('|:-:|:-:|:-:|:-:|:-:|:-:|:-:|:-:|:-:|')
    for rank in range(7, -1, -1):
        row = [f'**{rank + 1}**']
        for file in range(8):
            sq = chess.square(file, rank)
            piece = board.piece_at(sq)
            row.append(PIECES[piece.symbol()] if piece else '·')
        lines.append('| ' + ' | '.join(row) + ' |')
    return '\n'.join(lines)

def update_readme(board, last_move):
    with open('README.md', 'r') as f:
        content = f.read()

    new_board = board_to_markdown(board)
    turn = "Black (me — I'll respond soon!)" if board.turn == chess.BLACK else "White (you)"

    board_section = f"""<!-- CHESS BOARD START -->
<!-- The board below updates via GitHub Actions. To initialize, follow setup in the workflow section. -->

<div align="center">

{new_board}

</div>

**⬆️ Last move: `{last_move}` · Now playing: {turn}**"""

    updated = re.sub(
        r'<!-- CHESS BOARD START -->.*?(?=\*\*Make a move)',
        board_section + '\n\n',
        content,
        flags=re.DOTALL
    )

    with open('README.md', 'w') as f:
        f.write(updated)

def main():
    issue_title = sys.argv[1]
    move_str = issue_title.replace('chess: move ', '').strip()

    # Load or init board (in production, persist FEN in a file)
    board = chess.Board()

    try:
        move = chess.Move.from_uci(move_str)
        if move in board.legal_moves:
            board.push(move)
            # Simple AI: pick a random legal move
            import random
            if not board.is_game_over() and board.turn == chess.BLACK:
                ai_move = random.choice(list(board.legal_moves))
                board.push(ai_move)
            update_readme(board, move_str)
            print(f"✅ Move {move_str} applied.")
        else:
            print(f"❌ Illegal move: {move_str}")
    except Exception as e:
        print(f"❌ Error: {e}")

if __name__ == '__main__':
    main()
```

> 💡 **Tip:** For a persistent board state (not resetting on every push), save the FEN string to `.github/chess_state.txt` and load it in the script.
